Indexing and Swizzling for Vec

Examples

use beagle::vec::{Vec3,Vec2};
use beagle::index::swizzle::xyzw::*;

let mut v = Vec3::new([8f32, 12f32, 10f32]);
v[YZ] = v[XY] + Vec2::from_value(1f32);
assert_eq!(v, Vec3::new([8f32, 9f32, 13f32]));
